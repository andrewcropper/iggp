  (<= (empty ?c ?h)
      (true (cell ?c ?h b)))
      
  (<= (filled ?c ?h)
      (true (cell ?c ?h dirt)))
  (<= (filled ?c ?h)
      (true (cell ?c ?h w)))
  (<= (filled ?c ?h)
      (true (cell ?c ?h r)))
      
  (<= (next (cell ?c ?h2 w))
      (does white (drop ?c))
      (true (cell ?c ?h2 b))
      (filled ?c ?h1)
      (succ ?h1 ?h2))
    
  (<= (next (cell ?c ?h2 r))
      (does red (drop ?c))
      (true (cell ?c ?h2 b))
      (filled ?c ?h1)
      (succ ?h1 ?h2))

  (<= (next (cell ?x ?y ?z))
      (true (cell ?x ?y ?z))
      (distinct ?z b))

  (<= (next (cell ?c2 ?y b))
      (does ?w (drop ?c1))
      (true (cell ?c2 ?y b))
      (distinct ?c1 ?c2))

  (<= (next (control white))
      (true (control red)))

  (<= (next (control red))
      (true (control white)))

  ;; horizontal
  (<= (row ?z) (true (cell ?x1 ?y ?z))
               (distinct ?z b)
               (distinct ?z dirt)
               (succ ?x1 ?x2)
               (true (cell ?x2 ?y ?z))
               (succ ?x2 ?x3)
               (true (cell ?x3 ?y ?z))
               (succ ?x3 ?x4)
               (true (cell ?x4 ?y ?z))
               )
              
  ;; vertical
  (<= (column ?z) (true (cell ?x ?y1 ?z))
                  (distinct ?z b)
                  (succ ?y1 ?y2)
                  (true (cell ?x ?y2 ?z))
                  (succ ?y2 ?y3)
                  (true (cell ?x ?y3 ?z))
                  (succ ?y3 ?y4)
                  (true (cell ?x ?y4 ?z))
                  )
               
  ;; diagonal (north-east)
  (<= (diag ?z) (true (cell ?x1 ?y1 ?z))
                (distinct ?z b)
                (succ ?x1 ?x2)
                (succ ?y1 ?y2)
                (true (cell ?x2 ?y2 ?z))
                (succ ?x2 ?x3)
                (succ ?y2 ?y3)
                (true (cell ?x3 ?y3 ?z))
                (succ ?x3 ?x4)
                (succ ?y3 ?y4)
                (true (cell ?x4 ?y4 ?z))
                )
                
  ;; diagonal (south-east)
  (<= (diag ?z) (true (cell ?x1 ?y1 ?z))
                (distinct ?z b)
                (succ ?x1 ?x2)
                (succ ?y2 ?y1)
                (true (cell ?x2 ?y2 ?z))
                (succ ?x2 ?x3)
                (succ ?y3 ?y2)
                (true (cell ?x3 ?y3 ?z))
                (succ ?x3 ?x4)
                (succ ?y4 ?y3)
                (true (cell ?x4 ?y4 ?z))
                )
 
  (<= (line ?x) (row ?x))
  (<= (line ?x) (column ?x))
  (<= (line ?x) (diag ?x))
  
  (<= open
      (empty ?c ?h))
  
  (<= terminal
      (line r))

  (<= terminal
      (line w))

  (<= terminal
      (not open))

  (<= (legal ?w (drop ?c))
      (true (cell ?c ?y2 b))
      (filled ?c ?y1)
      (succ ?y1 ?y2)
      (true (control ?w)))

  (<= (legal white noop)
      (true (control red)))

  (<= (legal red noop)
      (true (control white)))
      
  (<= (goal white 100)
      (line w))

  (<= (goal white 50)
      (not (line r))
      (not (line w))
      (not open))

  (<= (goal white 0)

  (<= (goal red 100)

      (not open))

  (<= (goal red 0)
      (line w))

      (role ?w)
