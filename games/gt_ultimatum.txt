;;;
;;;  Game Theory: The Iterated Ultimatum Game
;;;
;;;  A two player game with alternating play in which, in each round,
;;;  the first player picks a way to split 10 points between both players.
;;;  The second player then picks whether or not to accept the split.
;;;  If the second player accepts, both players receive the points;
;;;  otherwise, neither player receives anything.
;;;
;;;  Background: http://en.wikipedia.org/wiki/Ultimatum_game
;;;
;;;  GDL BY: Sam Schreiber (schreib@cs.stanford.edu)
;;;
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;;; ROLE Relations
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(role white)
(role black)

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; Constants
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(maxRounds 10)

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; INIT Relations
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(init (whiteScore 0))
(init (blackScore 0))
(init (rounds 0))
(init (control white))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; LEGAL Relations
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(<= (legal white (offer ?x1))
    (index ?x1)
    (true (control white)))

(<= (legal black accept)
    (true (control black)))

(<= (legal black reject)
    (true (control black)))

(<= (legal white noop)
    (true (control black)))

(<= (legal black noop)
    (true (control white)))

legal(black,noop):-true_control(white)

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; NEXT Relations
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(<= (next (rounds ?x1))
    (true (rounds ?x))
    (true (control black))
    (sum ?x 1 ?x1))

(<= (next (rounds ?x))
    (true (rounds ?x))
    (true (control white)))

(<= (next (offered ?x1 ?x2))
    (sum ?x1 ?x2 10)
    (does white (offer ?x2)))

(<= (next (whiteScore ?xNext))
    (true (whiteScore ?x))
    (does black accept)
    (true (offered ?score ?dontcare))
    (sum ?x ?score ?xNext))

(<= (next (blackScore ?xNext))
    (true (blackScore ?x))
    (does black accept)
    (true (offered ?dontcare ?score))
    (sum ?x ?score ?xNext))

(<= (next (whiteScore ?x))
    (true (whiteScore ?x))
    (does black reject))

(<= (next (blackScore ?x))
    (true (blackScore ?x))
    (does black reject))

(<= (next (whiteScore ?x))
    (true (whiteScore ?x))
    (true (control white)))

(<= (next (blackScore ?x))
    (true (blackScore ?x))
    (true (control white)))

(<= (next (control black))
    (true (control white)))

(<= (next (control white))
    (true (control black)))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; TERMINAL Relations
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(<= terminal
    (true (rounds ?x))
    (maxRounds ?x))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; GOAL Relations
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(<= (goal white ?x)
    (true (whiteScore ?x)))

(<= (goal black ?x)
    (true (blackScore ?x)))

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
;; View Definitions
;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

(<= (sum ?x 0 ?x)
    (succ ?x ?dontcare))

(<= (sum ?xn 1 ?x)
    (succ ?xn ?x))

(<= (sum ?xn ?y ?x)
    (index ?y)
    (succ ?yless ?y)
    (succ ?ylessless ?yless)
    (succ ?x1 ?x)
    (sum ?xn ?yless ?x1))
